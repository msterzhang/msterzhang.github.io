1.requests库的使用

import requests
import re

def get_img():
    url = 'http://jandan.net/ooxx'
    html = requests.get(url).text
    find_img_list = re.findall('img src="(.*)",html')
    for img in find_img_list:
        print(img)

if __name__ == '__main__':
    get_img()
