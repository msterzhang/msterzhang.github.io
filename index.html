#!/usr/bin/env python
# -*- coding: utf-8 -*-
import json
from urllib import parse
import requests
import os
import time
from bs4 import BeautifulSoup as BS
# 存入题库，自行决定，是否启用
# import sqltiku


def get_appinf(filename):
    f = open(filename, 'r', encoding='utf-8')
    try:
        j = json.loads(f.read())
        if 'quiz' in j['data'] and 'options' in j['data']:
            num = j['data']['num']
            quiz = j['data']['quiz']
            print(('第' + str(num) + '题：' + quiz).center(50, '*') + '\n')
            cho = j['data']['options']
            lei = j['data']['school']
            ke = j['data']['type']
            zuo = j['data']['contributor']
            ti = quiz
            a = cho[0]
            b = cho[1]
            c = cho[2]
            d = cho[3]
            find_data = '无'
            # 存入数据库
            # sqltiku.find_sql(lei, ke, zuo, ti, a, b, c, d, find_data)
        else:
            pass
        return quiz, cho

    except:
        pass
    f.close()


def search(question, choice):
    pagenum = [0, 10, 20]
    answer = []
    for i in pagenum:
        q = parse.quote(question.encode('gbk'))
        url = 'https://zhidao.baidu.com/search?word=' + q + '&ie=gbk&site=-1&sites=0&date=0&pn=' + str(i)
        requests.packages.urllib3.disable_warnings()  # 忽视网页安全性问题
        r = requests.get(url, verify=False)  # 不验证证书
        r.encoding = 'gbk'
        soup = BS(r.text, 'html.parser')
        want = soup.find('div', id='wgt-list')
        wants = want.find_all('dl', class_='dl')
        for i in wants:
            ans = i.find('dd', class_='dd answer').text
            answer.append(ans)

    choiceset = {}
    choiceset['A'] = choice[0]
    choiceset['B'] = choice[1]
    choiceset['C'] = choice[2]
    choiceset['D'] = choice[3]

    # 计算四个选项在爬取百度答案中的出现次数
    results = {}
    for i in choiceset:
        account = []
        for j in answer:
            if choiceset[i] in j:
                account.append(j)
        result = len(account) / 30
        results[i] = result
        if i == 'D':
            print(('选' + i + '的可能性是：%.2f%%' % (result * 100)).center(50) + '\n')
        else:
            print(('选' + i + '的可能性是：%.2f%%' % (result * 100)).center(50))

    bestchoice = max(results.items(), key=lambda x: x[1])[0]
    print(('此题最好选' + bestchoice).center(50, '-') + '\n\n\n')

    # 点击答案，每个手机分辨率不同，自行调整
    if bestchoice == 'A':
        os.system('adb shell input swipe 528 1045 528 1045')
    elif bestchoice == 'B':
        os.system('adb shell input swipe 528 1255 528 1255')
    elif bestchoice == 'C':
        os.system('adb shell input swipe 528 1432 528 1432')
    elif bestchoice == 'D':
        os.system('adb shell input swipe 528 1637 528 1637')


def main():
    try:
        # 你在抓包程序中填的目录
        que, cho = get_appinf(r'E:\python\wchatgame\responsebody.txt')
        search(que, cho)

    except:
        # pass
        # 点击再来一次
        os.system('adb shell input swipe 506 1363 506 1363')
        # 选择挑战种类，这里你得自己调整啦
        os.system('adb shell input swipe 526 759 526 759')


if __name__ == '__main__':
    print('程序运行中...........')
    while True:
        main()
        time.sleep(1)

import re
from requests_html import HTMLSession
session = HTMLSession()


def get_img():
    r = session.get('http://jandan.net/ooxx')
    r.html.render()
    html = r.html.html
    img_list = re.findall('img src="(.*?)" style="max-width: 480px; ',html)
    for img_url in img_list:
        print(img_url)


if __name__ == '__main__':
    get_img()

钢结构笔记

第一章 绪论

钢结构优点：

  • 强度高，塑形，韧性好
  • 钢结构重量轻
  • 材质均匀，和力学计算的假定符合

钢结构缺点：

  • 耐腐蚀性差
  • 耐热不耐火
  • 可能发生脆性破坏

钢结构设计方法：

  • 以概率理论为基础的极限状态设计方法

第二章 钢结构的材料

钢结构对材料的要求：

  • 1.较高的强度：较高的抗拉强度fu和屈服点fy。
  • 2.足够的变形能力：较好的塑形和韧性。
  • 3.良好的工艺性能：冷加工，热加工和可焊性。

钢材的破坏：

  • 塑性破坏：由于形变过大，超过了材料或者构件可能的应变能力而产生的。
  • 脆性破坏：脆性破坏前塑形变形很小，甚至没有塑形变形，计算应力可能小于钢材的屈服点fy。

钢材主要性能：

  • 设计中以下屈服点为依据
  • 抗拉强度fu成为材料的强度储备
  • G=E/2(1+Y) #y为泊松比
  • 冲击韧性是钢材抵抗冲击荷载的能力，是钢材划分质量等级的依据

可焊性：

  • 可焊性:可焊性是指钢材通过普通的焊接工艺便可以得到合格的焊缝的性能。

温度对于钢材强度的影响:

  • 温度对于钢材强度的影响的趋势是随着温度升高，强度降低，应变增大；反之，随着温度的降低，强度略有增加，随着塑形和韧性的降低而变脆。

应力集中：

  • 当发生应力集中时候，钢材会变脆。

为什么在计算时候不考虑钢材的应力集中？

  • 由于钢材的塑性较好，在一定程度上使应力进行重分配，应力分布严重不均匀的现象趋于平缓。

反复荷载：

  • 在反复的荷载作用下，钢材会发生疲劳破坏，疲劳破坏属于脆性破坏

钢材疲劳计算：

  • 容许应力幅法
  • 荷载采用标准值
  • 容许应力幅与钢种无关

第三章 钢结构的连接

钢结构连接的基本原则：

  • 钢结构的连接必须符合安全可靠，传力明确，构造简单，制造方便和节约钢材的原则，

连接方式有：

  • 焊接连接，铆钉连接，螺栓连接。其中，焊接连接是现代钢结构中最主要的连接方式

文本转音频并自动播放

这个项目可以批量将文本转为音频，并且支持一般播放器播放，这里我们使用了python的一个库来播放，主要是源于我自己做的一个语音聊天机器人

注意：使用过程中如果报错以下错，请重新安装python-magic-bin

It is better to raise an ImportError since we are importing magic module raise
ImportError('failed to find libmagic.  Check your installation')

代码：

# encoding=utf-8
import requests
import pygame
import time
import eyed3

# 使用pygame播放的话需要设置歌曲名为非中文，使用如网易云这类的播放器就不用管这个
path = 'music.mp3'

# 转为音频文件
def get_mp3():
    text = '''
    央视网消息：近期，中央政治局同志首次向党中央和习近平总书记书面述职，这是他们履新近半年来交的首份答卷。习近平认真审阅了中央政治局同志的述职报告，并就中央政治局同志履行职责、做好工作、改进作风提出重要要求。
    '''
    url = 'https://tts.baidu.com/text2audio?cuid=baiduid&lan=zh&ctp=1&pdt=311&spd=6&tex='+text
    find = requests.get(url)
    with open(path,'wb') as f:
        f.write(find.content)
        f.close()

# 播放音频
def player():
    pygame.mixer.init()
    print("播放音乐")
    pygame.mixer.music.load(path)
    xx = eyed3.load(path)
    music_time = xx.info.time_secs
    pygame.mixer.music.play()
    time.sleep(music_time)
    pygame.mixer.music.stop()

if __name__ == '__main__':
    get_mp3()
    player()
