土木工程施工

一.概述

1.土方与基础工程

习题应用案例:

1.用于已知道垫层尺寸的土方挖方量计算：

代码:

#!/usr/bin/env python
# -*- coding: utf-8 -*-
# 用于已知道垫层尺寸的土方计算


def get_number():
    print(80*'=)
    # 输入相关参数
    print('欢迎使用土方计算程序 ---张某')
    print(80*'=')
    v_j = float(input('请输入基础体积(单位：m³)：'))
    l = float(input('请输入垫层的长度(单位：m)：'))
    b = float(input('请输入垫层的宽度(单位：m)：'))
    h = float(input('请输入基坑深(单位：m)：'))
    kp_1 = float(input('请输入可松性系数Kp：'))
    kp_2 = float(input("请输入可松系数Kp'："))
    f = float(input('请输入a+b+c的值(单位：m)：'))
    i = input('请输入放坡(如1:0.75，输入为1/0.75)：')
    i_find = float(i.strip('/')[2:])

    # 基坑相关面积计算
    A_1 = (b+2*(f+i_find*h))*(l+2*(f+i_find*h))
    A_2 = (b+2*f)*(l+2*f)
    A_0 = (b+2*(f+0.5*i_find*h))*(l+2*(f+0.5*i_find*h))

    # 基坑开挖量计算
    v_1 = h*(A_1+A_2+4*A_0)/6
    v_2 = v_1*kp_1
    v_3 = v_1-v_j
    v_1_1 = v_3/kp_2
    v_2_2 = v_1_1*kp_1
    v_s = v_2-v_2_2

    # 打印结果
    print(80*'=')
    print('土方开挖量：%d m³' % v_1)
    print('需要的填土量：%d m³' % v_2_2)
    print('运走的土量：%d m³' % v_s)
    print(80*'=')
    input('输入任意键结束计算')

if __name__ == '__main__':
    get_number()

2.条形基坑土方计算

代码:

#!/usr/bin/env python
# -*- coding: utf-8 -*-
# 用于已知道条形基础的土方计算


def get_number():
    print(80*'=')
    print('欢迎使用土方计算程序 ---张某')
    print(80*'=')
    v_j = float(input('请输入基础体积(单位：m³)：'))
    l = float(input('请输入基坑的长度(单位：m)：'))
    b = float(input('请输入基坑的宽(单位：m)：'))
    h = float(input('请输入基坑深(单位：m)：'))
    kp_1 = float(input('请输入可松性系数Kp：'))
    kp_2 = float(input("请输入可松系数Kp'："))
    i = input('请输入放坡(如1：0.75，输入为1/0.75)：')
    i_find = float(i.strip('/')[2:])

    # 基坑相关面积计算
    A_1 = 0.5*((b+2*i_find*h)+b)*h

    # 基坑开挖量计算
    v_1 = A_1*l
    v_2 = v_1*kp_1
    v_3 = v_1-v_j
    v_1_1 = v_3/kp_2
    v_2_2 = v_1_1*kp_1
    v_s = v_2-v_2_2

    # 打印结果
    print(80*'=')
    print('计算结果:')
    print('土方开挖量：%d m³' % v_1)
    print('需要的填土量：%d m³' % v_2_2)
    print('运走的土量：%d m³' % v_s)
    print(80*'=')
    input('输入任意键结束计算')


if __name__ == '__main__':
    get_number()
