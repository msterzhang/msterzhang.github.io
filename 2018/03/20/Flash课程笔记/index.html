Flask课程笔记

第一课 Flask学习之网页计算器，主要学习内容是表单的提交

注意：

  • 1.css,js,img这些文件应该放在main.py目录下新建一个叫static的文件夹存放
  • 2.html文件应该放在main.py目录下新建一个叫templates的文件夹存放

python文件main.py

!/usr/bin/env python
# -*- coding: utf-8 -*-
from flask import Flask,url_for
from flask import render_template
from flask import request
from flask import redirect
app = Flask(__name__)


# 访问主页时跳转到add
@app.route('/',methods=['GET','POST'])
def index():
    return redirect(url_for('add'))


@app.route('/add',methods=['GET','POST'])
def add():
    print(request.method)
    if request.method == 'POST':
        a = request.form['adder-1']
        b = request.form['adder-2']
        massage = int(a) + int(b)
        url_for('static', filename='style.css')
        return render_template('index.html', massage=massage,a=a,b=b)
    return render_template('index.html')

if __name__ == '__main__':
    app.run()

            <input type="text" readonly="readonly" placeholder="result" name="result" value="{{massage}}">
            <input type="submit" value="计算" onclick="add()">
        </form>
